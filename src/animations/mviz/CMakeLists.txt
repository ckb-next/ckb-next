project(mviz C)

find_package(PulseAudioSimple REQUIRED)

add_executable(mviz)

target_sources(
    mviz
        PRIVATE
          main.c)

target_include_directories(
    mviz
        PRIVATE
          ${PULSEAUDIOSIMPLE_INCLUDE_DIRS})

target_link_libraries(
    mviz
        PRIVATE
          kissfft
          m
          ${PULSEAUDIOSIMPLE_LIBRARIES})

install(
    TARGETS mviz
    DESTINATION ${INSTALL_DIR_ANIMATIONS})

add_subdirectory(kissfft)
